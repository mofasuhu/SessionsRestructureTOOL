\begin{question}
Podemos utilizar lo/la/los/las en vez del objeto directo\vspace*{20pt}

	
\begin{center}
\measureme{\renewcommand{\arraystretch}{1.5}\setlength{\tabcolsep}{30pt}\begin{tabular}{|c|c|c|}\hline
  & \textbf{Singular} & \textbf{Plural}   \\\hline
\textbf{Masculino } & Lo   & Los  \\\hline
\textbf{Femenino} & La  & Las  \\\hline
\end{tabular}}
\end{center}\vspace*{20pt}

Ejemplos


\begin{center}
\measureme{\renewcommand{\arraystretch}{1.2}\setlength{\tabcolsep}{20pt}\begin{tabular}{ll}
\textit{El fan difunde \textbf{una} autobiografía.}  & \textit{El fan \underline{la} difunde.}   \\
\textit{El fan difunde autobiografías.}  & \textit{El fan \underline{las} difunde.}   \\
\textit{Inés graba \textbf{un} concierto.}  & \textit{Inés \underline{lo} graba.} \\
\textit{Inés graba conciertos.}
  & \textit{Inés \underline{los} graba.}   \\
\end{tabular}}
\end{center}
\end{question}
